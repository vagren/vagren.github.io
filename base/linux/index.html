RPM #

rpm， Red-Hat Package Manage/RPM Package Manager

# install
rpm -ivh FileName.rpm

# query
rpm -ql PackageName
rpm -qi PackageName
rpm -qc PackageName

# update
rpm -Uvh FileName.rpm

# erase（卸载）
rpm -e PackageName
